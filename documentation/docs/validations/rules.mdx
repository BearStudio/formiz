### isRequired()

Test if there is value.

```jsx
import { isRequired } from '@formiz/validations'

<MyField
  name="fieldName"
  validations={[
    {
      rule: isRequired(),
      message: 'This field is required',
    },
  ]}
>
```

### isNotEmptyString()

```jsx
import { isNotEmptyString } from '@formiz/validations'

  name="fieldName"
  validations={[
    {
      rule: isNotEmptyString(),

    },
  ]}
>
```
